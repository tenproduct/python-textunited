- en_ca = 158  # English (Canada)
- en_gb = 40  # English (UK)
- en_us = 41  # English (US)
- es_co = 108  # Spanish (Colombia)
- es_es = 104  # Spanish (Spain)
- fr_ca = 48  # French (Canada)
- ja = 72  # Japanese
- ko = 76  # Korean
- pt_br = 94  # Portuguese (Brazil)
- zh_hant = 33  # Chinese (Traditional)
- zn_hans = 32  # Chinese (Simplified)
